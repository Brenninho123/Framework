const app = new ProjectJS({

    el: '#app',

    data: {
        count: 0
    },

    template: `
        <h1>{{ count }}</h1>

        <button pj-click="add">
            Add
        </button>
    `,

    methods: {
        add() {
            this.data.count++
        }
    }
})
